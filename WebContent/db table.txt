CREATE TABLE  "NEWACCOUNT" 
   (	"id" NUMBER, 
	"username" VARCHAR2(4000), 
	"password" VARCHAR2(4000), 
	"repassword" VARCHAR2(4000), 
	"amount" NUMBER, 
	"adderess" VARCHAR2(4000), 
	"phone" NUMBER, 
	 CONSTRAINT "NEWACCOUNT_PK" PRIMARY KEY ("id") ENABLE
   )
/


CREATE OR REPLACE TRIGGER  BI_NEWACCOUNT 
  BEFORE INSERT ON NEWACCOUNT               
  FOR EACH ROW  
BEGIN   SELECT LPAD(LTRIM(RTRIM(TO_CHAR(BANK_SEQ.NEXTVAL))),10,'0')  INTO :NEW.id FROM DUAL;
     
END; 

/
ALTER TRIGGER  "BI_NEWACCOUNT" ENABLE
/

CREATE SEQUENCE SYSTEM.BANK_SEQ
  START WITH 161
  MAXVALUE 9999999999999
  MINVALUE 1
  NOCYCLE
  CACHE 20
  NOORDER;
